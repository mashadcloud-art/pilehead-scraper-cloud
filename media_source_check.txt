// -- Saved config (persisted to localStorage) --------------
const CFG_KEY = 'ph_config';
function loadCfg() {
  try { return JSON.parse(localStorage.getItem(CFG_KEY) || '{}'); } catch(e) { 
return {}; }
}
function saveCfg(patch) {
  const cfg = { ...loadCfg(), ...patch };
  localStorage.setItem(CFG_KEY, JSON.stringify(cfg));
  return cfg;
}

function loadCfgFile() {
  try {
    const p = path.join(__dirname, 'config', 'settings.json');
    if (!fs.existsSync(p)) return {};
    const raw = fs.readFileSync(p, 'utf8');
    return JSON.parse(raw || '{}');
  } catch(e) { return {}; }
}

function stripQuotes(v) {
  if (typeof v !== 'string') return v || '';
  return v.replace(/^["']+|["']+$/g, '');
}

function mapFileCfg(c) {
  if (!c || typeof c !== 'object') return {};
  return {
    wpUrl: c.wp?.url || '',
    wpKey: c.wp?.key || '',
    wpSecret: c.wp?.secret || '',
    autoUpload: !!c.wp?.autoUpload,
    azEndpoint: c.azure?.endpoint || '',
    azDeployment: c.azure?.deployment || '',
    azKey: c.azure?.apiKey || '',
    azVersion: c.azure?.apiVersion || '',
    gcsBucket: c.gcs?.bucket || '',
    gcsToken: c.gcs?.token || '',
    gcsKeyPath: stripQuotes(c.gcs?.serviceAccountPath || ''),
    gcsImgFolder: c.gcs?.imageFolder || '',
    gcsDsFolder: c.gcs?.folder || '',
    gcsFosrocImgFolder: c.gcs?.fosrocImageFolder || '',
    gcsPublicRead: !!c.gcs?.publicRead,
    gcsImageStorage: c.gcs?.imageStorage || 'gcs'
  };
}

function toScraperConfig(c) {
  return {
    wp: {
      url: c.wpUrl || '',
      key: c.wpKey || '',
      secret: c.wpSecret || '',
      autoUpload: !!c.autoUpload
    },
    azure: {
      endpoint: c.azEndpoint || '',
      apiKey: c.azKey || '',
      deployment: c.azDeployment || '',
      apiVersion: c.azVersion || ''
    },
    gcs: {
      bucket: c.gcsBucket || '',
      token: c.gcsToken || '',
      serviceAccountPath: c.gcsKeyPath || '',
      folder: c.gcsDsFolder || '',
      imageFolder: c.gcsImgFolder || '',
      fosrocImageFolder: c.gcsFosrocImgFolder || '',
      publicRead: !!c.gcsPublicRead,
      imageStorage: c.gcsImageStorage || 'gcs'
    }
  };
}

function buildWpLinks(product, cfg) {
  if (!product || !cfg || !cfg.wpUrl) return { view: '', edit: '' };
  const base = cfg.wpUrl.replace(/\/+$/g, '');
  const view = product.permalink ? product.permalink : (product.id ? 
`${base}/?p=${product.id}` : '');
  const edit = product.id ? 
`${base}/wp-admin/post.php?post=${product.id}&action=edit` : '';
  return { view, edit };
}

// Pre-fill settings fields from saved config on load
function populateSettings() {
  const stored = loadCfg();
  const fileCfg = mapFileCfg(loadCfgFile());
  const merged = { ...fileCfg, ...stored };
  if (JSON.stringify(merged) !== JSON.stringify(stored)) saveCfg(merged);
  const c = loadCfg();
  const set = (id, val) => { const el = document.getElementById(id); if(el && 
val) el.value = val; };
  set('cfg-wp-url',        c.wpUrl);
  set('cfg-wp-key',        c.wpKey);
  set('cfg-wp-secret',     c.wpSecret);
  set('cfg-az-endpoint',   c.azEndpoint);
  set('cfg-az-deployment', c.azDeployment);
  set('cfg-az-key',        c.azKey);
  set('cfg-az-version',    c.azVersion);
  set('cfg-gcs-bucket',    c.gcsBucket);
  set('cfg-gcs-token',     c.gcsToken);
  set('cfg-gcs-keypath',   c.gcsKeyPath);
  set('cfg-gcs-img-folder',c.gcsImgFolder);
  set('cfg-gcs-ds-folder', c.gcsDsFolder);
  const imgStorageEl = document.getElementById('cfg-gcs-image-storage');
  if (imgStorageEl) imgStorageEl.value = c.gcsImageStorage || 'gcs';
  const autoEl = document.getElementById('auto-upload-check');
  if (autoEl) autoEl.checked = !!c.autoUpload;
}

let scraping=false, stimer=null, step=0, urls=[];

function setProgress(pct, label='') {
  document.getElementById('prog-fill').style.width=pct+'%';
  document.getElementById('prog-pct').textContent=pct+'%';
  document.getElementById('prog-label').textContent=label||'Processing…';

  const raw=document.getElementById('url-input').value.trim();

  if(!list.length){ toast('Paste at least one URL','warn'); sysLog('Start 
failed — no URLs','WARN'); return; }
  saveUrlHistory(list);
  scraping=true; urls=list; step=0; S.queue=list.length;
  document.getElementById('start-btn').disabled=true;
  document.getElementById('stop-btn').disabled=false;
  document.getElementById('prog-fill').classList.add('running');
  document.getElementById('q-badge').textContent=`${list.length} Tasks`;
  syncStats();
  const tbody=document.getElementById('tasks-body');
  tbody.innerHTML='';
  const selectedSource = websiteSelect ? websiteSelect.value : 'auto';
  list.forEach((url,i)=>{
    const tr=document.createElement('tr');
    tr.id = `row-${i}`;
    tr.innerHTML=`
      <td class="mono" style="max-width:190px;overflow:hidden;text-overflow:ell
ipsis;white-space:nowrap;" title="${url}">${url}</td>
      <td>${selectedSource}</td>

    if (openBtn) {
      openBtn.addEventListener('click', () => { try { shell.openExternal(url); 
} catch(_) {} });
    }
  });
  sysLog(`Scrape started — ${list.length} URL(s) queued`,'INFO');
  toast(`Scraping ${list.length} URL(s)`,'ok');
  runNext();
});

async function runNext() {
  if(!scraping || step >= urls.length){ finish(); return; }
  const i = step, url = urls[i];
  document.getElementById('st-'+i).innerHTML='<span class="badge 
b-proc">Processing</span>';
  setProgress(Math.round((i/urls.length)*100));
  S.queue = urls.length - i; syncStats();
  sysLog(`? [${i+1}/${urls.length}] ${url.substring(0,60)}…`,'DEBUG');

  try {
    const cfg = loadCfg();
    const selectedWebsite = websiteSelect ? websiteSelect.value : 'auto';
    const autoUpload = document.getElementById('auto-upload-check')?.checked;
    if (autoUpload && (!cfg.wpUrl || !cfg.wpKey || !cfg.wpSecret)) {
      sysLog('Auto-upload enabled but WP credentials missing', 'WARN');
      toast('WP credentials missing', 'warn');
    }
    const config = toScraperConfig({ ...cfg, autoUpload });
    const result = await ipcRenderer.invoke('scrape-url', { url, 
selectedWebsite, config, autoUpload });

    if(!scraping) return;

    if(result && result.success){
      S.scraped++;
      document.getElementById('st-'+i).innerHTML='<span class="badge 
b-ok">Done</span>';
      sysLog(`? Scraped: ${url.substring(0,55)}`,'OK');
      const product = result.uploadResult ? (result.uploadResult.product || 
result.uploadResult) : null;
      const gcsImages = result.uploadResult?.gcs_images || {};
      const gcsInfo = result.uploadResult?.gcs || {};
      const gcsUrl = gcsImages.main || gcsInfo.url || '';
      if (product) S.uploaded++;
      const act = document.getElementById('act-'+i);
      if (act) {
        let html = `<button class="btn btn-default btn-xs" 
data-action="open-target">Open</button>`;
        if (product) {
          html += `<button class="btn btn-default btn-xs" 
data-action="open-wp">WP</button>`;
          html += `<button class="btn btn-default btn-xs" 
data-action="edit-wp">Edit</button>`;
        }
        if (gcsUrl) {
          html += `<button class="btn btn-default btn-xs" 
data-action="open-gcs">GCS</button>`;
        }
        act.innerHTML = html;
        const { view, edit } = buildWpLinks(product, cfg);
        
act.querySelector('[data-action="open-target"]')?.addEventListener('click', () 
=> { try { shell.openExternal(url); } catch(_) {} });
        
act.querySelector('[data-action="open-wp"]')?.addEventListener('click', () => 
{ if (view) { try { shell.openExternal(view); } catch(_) {} } });
        
act.querySelector('[data-action="edit-wp"]')?.addEventListener('click', () => 
{ if (edit) { try { shell.openExternal(edit); } catch(_) {} } });
        
act.querySelector('[data-action="open-gcs"]')?.addEventListener('click', () => 
{ if (gcsUrl) { try { shell.openExternal(gcsUrl); } catch(_) {} } });
      }

      const ul = document.getElementById('link-list');
      if(ul.querySelector('.empty')) ul.innerHTML='';
      const li = document.createElement('li');
      li.innerHTML=`<a href="#" 
data-url="${url}">${url.substring(0,68)}${url.length>68?'…':''}</a>`;
      li.querySelector('a')?.addEventListener('click', (e) => { 
e.preventDefault(); try { shell.openExternal(url); } catch(_) {} });
      ul.appendChild(li);
      document.getElementById('scraped-count').textContent=`${ul.querySelectorA
ll('li').length} links`;

    } else {
      S.failed++;
      document.getElementById('st-'+i).innerHTML='<span class="badge 
b-fail">Failed</span>';
      sysLog(`? Failed: ${result?.error || url.substring(0,55)}`,'ERROR');
    }
  } catch(err) {
    S.failed++;
    document.getElementById('st-'+i).innerHTML='<span class="badge 
b-fail">Error</span>';
    sysLog(`? Exception: ${err.message}`,'ERROR');
  }
  syncStats(); step++; runNext();
}

function finish() {
  scraping=false; S.queue=0;
  setProgress(100,`Complete — ${S.scraped} scraped, ${S.failed} failed`);
  document.getElementById('prog-fill').classList.remove('running');
  document.getElementById('start-btn').disabled=false;
  document.getElementById('stop-btn').disabled=true;
  document.getElementById('q-badge').textContent='0 Tasks';
  syncStats();
  sysLog(`Scrape complete — ${S.scraped} products, ${S.failed} errors`,'INFO');
  toast(`Done! ${S.scraped} scraped, ${S.failed} failed`,'ok');
}

(function setupRetroGcsPage() {
    const scopeSel     = document.getElementById('retro-scope');

    const progLog      = document.getElementById('retro-progress-log');

    if (!scopeSel) return; // page not yet in DOM (should not happen)

    let loadedProducts = [];

    // -- Scope field visibility --
    scopeSel.addEventListener('change', () => {
        idsField.style.display   = scopeSel.value === 'ids'   ? '' : 'none';
        rangeField.style.display  = scopeSel.value === 'range' ? '' : 'none';
    });

    // -- Load Products --
    loadBtn.addEventListener('click', async () => {
        const c   = loadCfg();
        const url = c.wpUrl; const key = c.wpKey; const secret = c.wpSecret;
        if (!url || !key) { loadStatus.textContent = '? WP credentials missing 
in Settings'; return; }

        const maxN   = parseInt(maxInput.value) || 50;
        const scope  = scopeSel.value;

        loadBtn.disabled = true;
        loadStatus.textContent = 'Loading…';
        productList.innerHTML  = '';
        selCard.style.display  = 'none';
        loadedProducts = [];

        try {
            let allProds = [];
            // Fetch pages until we have enough
            let page = 1;
            while (allProds.length < maxN) {
                const need = Math.min(100, maxN - allProds.length);
                const res  = await ipcRenderer.invoke('wp-list-products', {
                    url, key, secret,
                    page,
                    perPage: need,
                    orderby: 'date',
                    order: 'desc'
                });
                if (!res.success || !res.data) break;
                allProds = allProds.concat(res.data.products || []);
                if (page >= (res.data.totalPages || 1)) break;
                page++;
            }

            // Client-side scope filtering
            if (scope === 'ids') {
                const ids = idsInput.value.split(',').map(s => 
parseInt(s.trim())).filter(n => !isNaN(n));
                allProds = allProds.filter(p => ids.includes(p.id));
            } else if (scope === 'range') {
                const parts = rangeInput.value.split('-').map(s => 
parseInt(s.trim()));
                if (parts.length === 2 && !isNaN(parts[0]) && 
!isNaN(parts[1])) {
                    allProds = allProds.filter(p => p.id >= parts[0] && p.id 
<= parts[1]);
                }
            } else if (scope === 'no-gcs') {
                allProds = allProds.filter(p => {
                    const hasMeta = p.meta_data && p.meta_data.some(m => m.key 
=== 'gcs_image_url' && m.value);
                    return !hasMeta;
                });
            }

            loadedProducts = allProds.slice(0, maxN);
            loadStatus.textContent = `${loadedProducts.length} product(s) 
loaded`;

            // Render product list
            productList.innerHTML = '';
            loadedProducts.forEach(p => {
                const mainImg = p.images && p.images[0] ? p.images[0].src : '';
                const hasMeta = p.meta_data && p.meta_data.some(m => m.key === 
'gcs_image_url' && m.value);
                const row = document.createElement('div');
                row.style.cssText = 
'display:flex;align-items:center;gap:10px;padding:8px 
10px;background:var(--bg-3);border-radius:6px;';

                    <input type="checkbox" data-id="${p.id}" checked 
style="width:15px;height:15px;cursor:pointer;accent-color:var(--teal);">

style="width:40px;height:40px;object-fit:cover;border-radius:4px;border:1px 
solid var(--border);">` : '<div style="width:40px;height:40px;background:var(--
bg-4);border-radius:4px;"></div>'}
                    <div style="flex:1;overflow:hidden;">

wrap;overflow:hidden;text-overflow:ellipsis;">#${p.id} — ${p.name || '(no 
name)'}</div>
                      <div 
style="font-size:10px;color:var(--fg-4);">${p.images ? p.images.length : 0} 
image(s) · ${hasMeta ? '<span style="color:var(--teal);">? GCS linked</span>' 
: '<span style="color:var(--amber);">No GCS</span>'}</div>
                    </div>`;
                productList.appendChild(row);
            });

        } catch (err) {
            loadStatus.textContent = '? ' + err.message;
        } finally {
            loadBtn.disabled = false;
        }
    });

    function updateSelCount() {
        const checked = 
productList.querySelectorAll('input[type=checkbox]:checked').length;
        selCount.textContent = `${checked} selected`;
    }
    productList.addEventListener('change', updateSelCount);
    selectAllBtn.addEventListener('click', () => { 
productList.querySelectorAll('input[type=checkbox]').forEach(cb => cb.checked 
= true); updateSelCount(); });

productList.querySelectorAll('input[type=checkbox]').forEach(cb => cb.checked 
= false); updateSelCount(); });

parseInt(cb.dataset.id));

            folder:             c.gcsDsFolder,

style="color:var(--red);">? GCS bucket not configured in Settings</span>'; 
progCard.style.display = ''; return; }

            progLog.scrollTop = progLog.scrollHeight;
            if (data.done !== undefined) {
                const pct = Math.round((data.done / (data.total || 1)) * 100);

                progCount.textContent = `${data.done} / ${data.total || 
selected.length}`;

uploaded</span>  <span style="color:var(--red);">? ${err} failed</span>`;

                progLog.scrollTop = progLog.scrollHeight;
            }
        } catch (err) {
            const line = document.createElement('div');
            line.style.color = 'var(--red)';
            line.textContent = '? IPC error: ' + err.message;
            progLog.appendChild(line);
        } finally {
            ipcRenderer.removeListener('gcs-bulk-progress', progressHandler);
            uploadBtn.disabled = false;
        }
    });
})();

    const count   = b.ids ? b.ids.length : (b.products ? b.products.length : 
0);
    const firstId = b.ids ? b.ids[0] : (b.products && b.products[0] ? 
b.products[0].id : '?');
    const lastId  = b.ids ? b.ids[b.ids.length-1] : (b.products && 
b.products[count-1] ? b.products[count-1].id : '?');
    const idRange = count > 1 ? `#${firstId} \u2013 #${lastId}` : 
`#${firstId}`;
    let statusHtml, dotColor, rowBg = '';
    if (b.status === 'running') {
      statusHtml = '<span style="color:var(--blue);">\u25B6 
Uploading\u2026</span>';
      dotColor = 'var(--blue)'; rowBg = 'background:rgba(10,132,255,0.07);';
    } else {
      statusHtml = '<span style="color:var(--fg-5);">Pending</span>';
      dotColor = '#555';
    }
    const el = document.createElement('div');
    el.id = `ms-batch-tile-${idx}`;
    el.style.cssText = `display:flex;align-items:center;gap:12px;padding:9px 
14px;border-bottom:1px solid var(--border);${rowBg}`;
    el.innerHTML = `
      <div style="width:24px;height:24px;border-radius:50%;border:1.5px solid $
{dotColor};display:flex;align-items:center;justify-content:center;font-size:10p
x;font-weight:700;color:${dotColor};flex-shrink:0;">${idx + 1}</div>
      <div style="flex:1;min-width:0;">
        <span style="font-size:12px;font-weight:600;">Batch ${idx + 1}</span>
        <span 
style="font-size:11px;color:var(--fg-4);margin-left:8px;">${count} 
product${count !== 1 ? 's' : ''}</span>
        <span 
style="font-size:10px;color:var(--fg-5);margin-left:8px;">${idRange}</span>
      </div>
      <div style="font-size:11px;white-space:nowrap;">${statusHtml}</div>`;
    const existing = document.getElementById(`ms-batch-tile-${idx}`);
    if (existing) existing.replaceWith(el); else batchListEl.appendChild(el);
  }

  function updateBatchSummary() {
    if (!batchSummaryEl) return;
    const total     = migrateBatchQueue.length;
    const done      = migrateBatchQueue.filter(b => b.status === 
'done').length;
    const remaining = total - done;
    const prods     = migrateBatchQueue.reduce((s, b) => s + (b.ids ? 
b.ids.length : (b.products ? b.products.length : 0)), 0);
    batchSummaryEl.textContent = `${done}/${total} batches done \u2014 
${remaining} remaining \u2014 ${prods} products total`;
  }

  async function prepareMigrationBatches() {
    const { url, key, secret } = (function(){ const c = loadCfg(); return { 
url: c.wpUrl||'', key: c.wpKey||'', secret: c.wpSecret||'' }; })();
    const scope     = scopeSel ? scopeSel.value : 'no-gcs';
    const maxPages  = 
parseInt(document.getElementById('ms-migrate-pages')?.value  || '10', 10);
    const batchSize = 
parseInt(document.getElementById('ms-migrate-batch-size')?.value || '20', 10);

    migrateBatchQueue = []; migrateBatchIndex = 0;
    if (batchListEl)    batchListEl.innerHTML = '';
    if (batchQueueCard) batchQueueCard.style.display = '';
    if (nextBatchBtn)   { nextBatchBtn.disabled = true; 
nextBatchBtn.textContent = '\u23F3 Scanning\u2026'; }

    logLine('\u23F3 Scanning for products to migrate\u2026', '');

    if (scope === 'ids') {
      // IDs mode: just split typed IDs into batches — no pre-fetch needed
      const idsRaw = document.getElementById('ms-migrate-ids')?.value || '';
      const ids = idsRaw.split(',').map(s => parseInt(s.trim(), 10)).filter(n 
=> !isNaN(n) && n > 0);
      if (!ids.length) { logLine('\u2717 No product IDs entered.', 
'var(--red)'); return; }
      for (let i = 0; i < ids.length; i += batchSize)
        migrateBatchQueue.push({ ids: ids.slice(i, i + batchSize), products: 
null, status: 'pending', ok: 0, failed: 0 });
      logLine(`\u2705 ${ids.length} IDs \u2192 ${migrateBatchQueue.length} 
batch${migrateBatchQueue.length > 1 ? 'es' : ''}. Batch 1 starting\u2026`, 
'var(--teal)');
    } else {
      // Fetch all pages, collect products
      let allProducts = [];

      // For unmigrated-only scope, auto-detect total pages from WP so we 
never miss products.
      // For "all" scope, respect the user-set maxPages cap.
      let pages = Math.max(1, Math.min(maxPages, 100));
      if (scope === 'no-gcs') {
        try {
          const r0 = await ipcRenderer.invoke('wp-list-products', { url, key, 
secret, page: 1, perPage: 1 });
          if (r0 && r0.success && r0.data.total) {
            const totalProducts = r0.data.total;
            pages = Math.ceil(totalProducts / 100); // fetch every page
            logLine(`\u23F3 ${totalProducts} total products \u2192 scanning 
all ${pages} pages\u2026`, '');
          }
        } catch(_) {}
      }

      for (let page = 1; page <= pages; page++) {
        const r = await ipcRenderer.invoke('wp-list-products', { url, key, 
secret, page, perPage: 100 }).catch(() => null);
        if (!r || !r.success) { logLine(`\u2717 Failed to fetch page 
${page}.`, 'var(--red)'); break; }
        const batch = r.data.products || [];
        if (!batch.length) break;
        if (scope === 'no-gcs') {
          batch.forEach(p => {
            const has = Array.isArray(p.meta_data) && p.meta_data.some(m =>
              (m.key === 'gcs_image_url' || m.key === 'gcs_gallery_urls' || 
m.key === 'gcs_no_images' || m.key === 'gcs_upload_failed') && m.value);
            if (!has) allProducts.push(p);
          });
        } else {
          allProducts = allProducts.concat(batch);
        }
        logLine(`\u23F3 Page ${page}/${pages} done \u2014 
${allProducts.length} unmigrated products found so far\u2026`, '');
        if (batch.length < 100) break;
      }
      if (!allProducts.length) {
        logLine('\u2713 All products already migrated \u2014 nothing to do!', 
'var(--green)');
        if (nextBatchBtn) { nextBatchBtn.disabled = true; 
nextBatchBtn.textContent = '\u2713 Nothing to migrate'; }
        if (batchSummaryEl) batchSummaryEl.textContent = '0 products to 
migrate';
        return;
      }
      for (let i = 0; i < allProducts.length; i += batchSize)
        migrateBatchQueue.push({ ids: null, products: allProducts.slice(i, i + 
batchSize), status: 'pending', ok: 0, failed: 0 });
      logLine(`\u2705 ${allProducts.length} unmigrated products \u2192 
${migrateBatchQueue.length} batch${migrateBatchQueue.length > 1 ? 'es' : ''} 
of \u2264${batchSize}. Batch 1 starting\u2026`, 'var(--teal)');
    }

    // Only render pending batches (skip done ones)
    migrateBatchQueue.forEach((b, i) => { if (b.status !== 'done') 
renderBatchTile(i); });
    updateBatchSummary();
  }

  async function runCurrentBatch() {
    if (migrateRunning || migrateBatchIndex >= migrateBatchQueue.length) 
return;
    const idx   = migrateBatchIndex;
    const batch = migrateBatchQueue[idx];
    batch.status = 'running'; renderBatchTile(idx);
    const tileEl = document.getElementById(`ms-batch-tile-${idx}`);
    if (tileEl) tileEl.scrollIntoView({ behavior: 'smooth', block: 'nearest' 
});

    const total = batch.ids ? batch.ids.length : batch.products.length;
    if (nextBatchBtn) { nextBatchBtn.disabled = true; nextBatchBtn.textContent 
= `\u25B6 Uploading batch ${idx + 1}\u2026`; }
    if (progressLog)  progressLog.innerHTML = '';
    if (progressCard) progressCard.style.display = '';
    if (progressBar)  progressBar.style.width = '0%';
    if (progressTag)  progressTag.textContent  = `Batch ${idx + 1} \u2014 0 / 
${total}`;
    migrateRunning = true;
    if (migrateBtn) migrateBtn.disabled = true;
    if (stopBtn)    stopBtn.disabled = false;

    ipcRenderer.on('media-source-progress', onMigrateProgress);
    try {
      const payload = batch.ids
        ? { scope: 'ids', productIds: batch.ids, maxPages: 1 }
        : { scope: 'preloaded', preloadedProducts: batch.products };
      const res = await ipcRenderer.invoke('media-source-migrate', payload);
      batch.ok     = (res && res.ok)     || 0;
      batch.failed = (res && res.failed) || 0;
      if (res && !res.success) logLine('\u2717 ' + (res.error || 'Unknown 
error'), 'var(--red)');
    } catch(e) {
      batch.ok = 0; batch.failed = total;
      logLine('\u2717 IPC error: ' + e.message, 'var(--red)');
    } finally {
      ipcRenderer.removeListener('media-source-progress', onMigrateProgress);
      migrateRunning = false;
      if (migrateBtn) { migrateBtn.disabled = false; migrateBtn.textContent = 
'\u2601 Start Migration'; }
      if (stopBtn)    stopBtn.disabled = true;
    }

    batch.status = 'done'; renderBatchTile(idx);
    migrateBatchIndex++;
    updateBatchSummary();

    const remaining = migrateBatchQueue.length - migrateBatchIndex;
    if (remaining > 0) {
      if (nextBatchBtn) {
        nextBatchBtn.disabled = false;
        nextBatchBtn.textContent = `\u25B6 Upload Batch ${migrateBatchIndex + 
1} of ${migrateBatchQueue.length}  (${remaining} remaining)`;
      }
      logLine(`\u2705 Batch ${idx + 1} done \u2014 ${batch.ok} ok, 
${batch.failed} failed.  Auto-starting next batch\u2026`, 'var(--teal)');
      // Auto-continue to the next batch
      await runCurrentBatch();
    } else {
      if (nextBatchBtn) { nextBatchBtn.disabled = true; 
nextBatchBtn.textContent = '\u2713 All batches complete!'; }
      logLine(`\uD83C\uDF89 All ${migrateBatchQueue.length} 
batch${migrateBatchQueue.length > 1 ? 'es' : ''} complete!`, 'var(--green)');
      // Auto-scan at 8s, 18s, 35s — WordPress needs time to index bulk meta 
updates
      logLine('\u23F3 Refreshing scan counts (auto in 8s, 18s, 35s)\u2026', 
'');
      const _scanDelays = [8000, 18000, 35000];
      _scanDelays.forEach(delay => setTimeout(runMigrateScan, delay));
    }
  }

  if (saveBtn) {

  if (testGcsBtn) {

  const migrateScanBtn    = document.getElementById('ms-migrate-scan-btn');

    const el = document.getElementById(id);

  async function runMigrateScan() {
    if (!migrateScanBtn) return;
    migrateScanBtn.disabled = true;
    migrateScanBtn.textContent = 'Scanning\u2026';
    if (migrateScanStatus) { migrateScanStatus.style.display = ''; 
migrateScanStatus.textContent = 'Fetching product count\u2026'; }

    const { url, key, secret } = (function(){ const c=loadCfg(); return 
{url:c.wpUrl||'',key:c.wpKey||'',secret:c.wpSecret||''}; })();
    const gcs = toScraperConfig(loadCfg()).gcs;
    if (!gcs.publicBase) gcs.publicBase = 'https://storage.googleapis.com/' + 
(gcs.bucket || '');

    // Step 1: get total product count quickly (1 per page)
    let wpTotal = 0;
    try {
      const r0 = await ipcRenderer.invoke('wp-list-products', { url, key, 
secret, page: 1, perPage: 1 });
      if (r0.success) wpTotal = r0.data.total || 0;
    } catch(_) {}

    setMStat('ms-mstat-wp-total', wpTotal || '\u2014', 'var(--blue)');
    if (migrateScanStatus) migrateScanStatus.textContent = 'Fetching all 
products to check GCS status (' + wpTotal + ' products)\u2026';

    // Step 2: fetch all product pages concurrently (100 per page) to check 
meta_data
    let withGcs = 0; let withoutGcs = 0;
    if (wpTotal > 0 && url) {
      const totalPages = Math.ceil(wpTotal / 100);
      const pageNums   = Array.from({ length: Math.min(totalPages, 50) }, (_, 
i) => i + 1); // cap at 50 pages = 5000 products
      if (migrateScanStatus) migrateScanStatus.textContent = 'Checking ' + 
pageNums.length + ' pages (' + wpTotal + ' products)\u2026';
      // batch 5 pages at a time
      for (let i = 0; i < pageNums.length; i += 5) {
        const batch = pageNums.slice(i, i + 5).map(p =>
          ipcRenderer.invoke('wp-list-products', { url, key, secret, page: p, 
perPage: 100 })
        );
        const results = await Promise.all(batch.map(p => p.catch(() => null)));
        for (const r of results) {
          if (!r || !r.success) continue;
          for (const p of (r.data.products || [])) {
            const hasGcs = Array.isArray(p.meta_data) && p.meta_data.some(m =>
              (m.key === 'gcs_image_url' || m.key === 'gcs_gallery_urls' || 
m.key === 'gcs_no_images' || m.key === 'gcs_upload_failed') && m.value
            );
            if (hasGcs) withGcs++; else withoutGcs++;
          }
        }
        if (migrateScanStatus) migrateScanStatus.textContent = 'Checked ' + 
Math.min((i + 5) * 100, wpTotal) + ' / ' + wpTotal + ' products\u2026';
      }
    }

    setMStat('ms-mstat-wp-gcs',     withGcs,    'var(--green)');
    setMStat('ms-mstat-wp-pending',  withoutGcs, withoutGcs > 0 ? 
'var(--gold)' : 'var(--green)');
    if (migrateScanStatus) migrateScanStatus.textContent = 'Fetching GCS file 
count\u2026';

    // Step 3: count GCS files
    let gcsTotal = '\u2014';
    try {
      const gr = await ipcRenderer.invoke('gcs-list-files', { gcsConfig: gcs, 
prefix: '', maxResults: 1000 });
      if (gr.success) {
        gcsTotal = (gr.data.items || []).length + (gr.data.nextPageToken ? '+' 
: '');
      }
    } catch(_) {}
    setMStat('ms-mstat-gcs-total', gcsTotal, 'var(--teal)');

    // Step 4: migration percentage
    const totalStr = typeof wpTotal === 'number' && wpTotal > 0;
    const pct = totalStr ? Math.round((withGcs / wpTotal) * 100) : 0;
    setMStat('ms-mstat-pct', totalStr ? pct + '%' : '\u2014', '#bf5af2');
    const pctBar = document.getElementById('ms-mstat-pct-bar');
    if (pctBar) pctBar.style.width = totalStr ? pct + '%' : '0%';

    if (migrateScanStatus) migrateScanStatus.textContent = '\u2713 Scan 
complete \u2014 ' + withGcs + ' / ' + wpTotal + ' products migrated to cloud';
    migrateScanBtn.disabled = false;
    migrateScanBtn.textContent = '\u21BB Scan Counts';
  }

  if (migrateScanBtn) migrateScanBtn.addEventListener('click', runMigrateScan);

  // -- Open GCS bucket in browser ------------------------------
  const openGcsBtn = document.getElementById('ms-open-gcs-btn');
  const openWpBtn  = document.getElementById('ms-open-wp-btn');

  if (openGcsBtn) {
    openGcsBtn.addEventListener('click', () => {
      const cfg    = loadCfg();
      const gc     = toScraperConfig(cfg).gcs;
      const bucket = gc.bucket;
      if (!bucket) { toast('GCS bucket not configured in Settings', 'warn'); 
return; }
      const url = 'https://console.cloud.google.com/storage/browser/' + bucket;
      shell.openExternal(url);
    });
  }

  if (openWpBtn) {
    openWpBtn.addEventListener('click', () => {
      const cfg = loadCfg();
      const url = cfg.wpUrl ? cfg.wpUrl.replace(/\/$/, '') + 
'/wp-admin/upload.php' : '';
      if (!url) { toast('WP URL not configured in Settings', 'warn'); return; }
      shell.openExternal(url);
    });
  }

  // (auto-scan is now triggered directly inside runCurrentBatch when all 
batches finish)

  // Reload status when the page becomes visible
  const pvEl = document.getElementById('pv-media-source');
  if (pvEl) {
    new MutationObserver(() => {
      if (pvEl.classList.contains('active') && pvEl.style.display !== 'none') 
loadStatus();
    }).observe(pvEl, { attributes: true, attributeFilter: ['style', 'class'] 
});
  }
  const sbBtn = document.querySelector('[data-sbnav="media-source"]');
  if (sbBtn) sbBtn.addEventListener('click', () => setTimeout(loadStatus, 
100));

})();

(function setupMediaManagerUI() {

  // -- core helpers --------------------------------------------
  function fmtBytes(b) {
    if (!b) return '0 B';
    const k = 1024, sizes = ['B','KB','MB','GB'];
    const i = Math.floor(Math.log(b) / Math.log(k));
    return (b / Math.pow(k, i)).toFixed(1) + ' ' + sizes[i];
  }
  function fmtDate(s) {
    if (!s) return '';
    return new Date(s).toLocaleDateString(undefined, { year:'2-digit', 
month:'short', day:'numeric' });
  }
  function wpCreds() {
    const cfg = loadCfg();
    return { url: cfg.wpUrl || '', key: cfg.wpKey || '', secret: cfg.wpSecret 
|| '' };
  }
  function gcsConf() {
    const sc = toScraperConfig(loadCfg());
    const g  = sc.gcs;
    if (!g.publicBase) g.publicBase = 'https://storage.googleapis.com/' + 
(g.bucket || '');
    return g;
  }

  // Run up to `size` promises concurrently.  tasks = array of () => Promise
  async function runBatch(tasks, size = 8) {
    const results = [];
    let i = 0;
    while (i < tasks.length) {
      const batch = tasks.slice(i, i + size).map(fn => fn().catch(e => ({ 
_err: e.message })));
      results.push(...(await Promise.all(batch)));
      i += size;
    }
    return results;
  }

  // Extract GCS image URLs stored in product meta_data
  function extractGcsUrls(product) {
    const urls = [];
    if (!product || !Array.isArray(product.meta_data)) return urls;
    for (const m of product.meta_data) {

m.value;
          if (Array.isArray(arr)) arr.forEach(u => u && urls.push(u));
        } catch(_) {}
      }
    }
    return urls;
  }

  // Delete a single product + optionally its WP media images + GCS images
  async function wppDeleteOne(url, key, secret, id, mediaIds, gcsUrls, addLog) 
{
    const tasks = [];
    // 1. delete WP product
    tasks.push(ipcRenderer.invoke('wp-delete-product', { url, key, secret, 
productId: id }));
    // 2. delete WP media (concurrent with product delete)
    if (wppDelMediaOpt() && mediaIds.length) {
      mediaIds.forEach(mid => tasks.push(
        ipcRenderer.invoke('wp-delete-media', { url, key, secret, mediaId: mid 
})
      ));
    }
    const [prodRes, ...mediaResults] = await Promise.allSettled(tasks);
    const prodOk = prodRes.status === 'fulfilled' && prodRes.value && 
prodRes.value.success;
    if (!prodOk) { addLog('\u2717 Product #' + id + ' failed: ' + 
(prodRes.value?.error || 'error'), 'var(--red)'); return false; }
    let log = '\u2713 #' + id;
    if (mediaIds.length && wppDelMediaOpt()) {
      const mOk = mediaResults.filter(r => r.status === 'fulfilled' && 
r.value?.success).length;
      log += '  [\u2713 ' + mOk + '/' + mediaIds.length + ' WP media]';
    }
    // 3. delete GCS images (after product is confirmed deleted)
    if (wppDelGcsOpt() && gcsUrls.length) {
      const gr = await ipcRenderer.invoke('gcs-delete-assets', { gcsConfig: 
gcsConf(), urls: gcsUrls });
      const gOk = (gr.data && gr.data.deleted && gr.data.deleted.length) || 0;
      log += '  [\u2713 ' + (typeof gOk === 'number' ? gOk : 
(gr.data?.deleted?.length || 0)) + '/' + gcsUrls.length + ' GCS]';
    }
    addLog(log, 'var(--green)');
    return true;
  }

      if (!confirm('Permanently delete ' + checked.length + ' media item(s) 
from WordPress?')) return;
      const { url, key, secret } = wpCreds();
      wpmDeleteBtn.disabled = true; wpmDeleteBtn.textContent = 'Deleting...';
      wpmLog.style.display = ''; wpmLog.innerHTML = '';
      let ok = 0; let fail = 0;
      const addLog = (m, c) => { const d = document.createElement('div'); 
if(c) d.style.color=c; d.textContent=m; wpmLog.appendChild(d); 
wpmLog.scrollTop=wpmLog.scrollHeight; };
      // 10 concurrent deletions
      const tasks = checked.map(cb => async () => {
        const id = parseInt(cb.dataset.id);
        const r  = await ipcRenderer.invoke('wp-delete-media', { url, key, 
secret, mediaId: id });
        if (r.success) { ok++; const tr = cb.closest('tr'); if (tr) 
tr.remove(); addLog('\u2713 Deleted #' + id, 'var(--green)'); }
        else           { fail++; addLog('\u2717 #' + id + ' ' + (r.error || 
'failed'), 'var(--red)'); }
      });
      await runBatch(tasks, 10);
      wpmItems = wpmItems.filter(m => !checked.some(c => String(m.id) === 
c.dataset.id));
      wpmCount.textContent = wpmItems.length;
      wpmUpdateSelCount();
      wpmDeleteBtn.disabled = false; wpmDeleteBtn.textContent = '\uD83D\uDDD1 
Delete Media';
      toast('\u2713 ' + ok + ' deleted' + (fail ? ', \u2717 ' + fail + ' 
failed' : ''), ok && !fail ? 'ok' : 'warn');
    });
  }

  // -------------------------------------------------------------
  //  STORAGE OVERVIEW — stats counts
  // -------------------------------------------------------------
  const statsRefreshBtn = document.getElementById('ms-stats-refresh-btn');
  const statsStatus     = document.getElementById('ms-stats-status');
  const statsTag        = document.getElementById('ms-stats-tag');

  function setStat(id, val, color) {
    const el = document.getElementById(id);
    if (!el) return;
    el.textContent = val;
    if (color) el.style.color = color;
  }

  async function statsLoad() {
    if (!statsRefreshBtn) return;
    statsRefreshBtn.disabled = true;
    statsRefreshBtn.textContent = 'Loading\u2026';
    if (statsStatus) statsStatus.textContent = 'Fetching GCS + WP 
counts\u2026';
    const { url, key, secret } = wpCreds();
    const gcs = gcsConf();

    const [gcsRes, wpImgRes, wpVidRes, wpDocRes] = await Promise.allSettled([
      ipcRenderer.invoke('gcs-list-files', { gcsConfig: gcs, prefix: '', 
maxResults: 1000 }),
      url ? ipcRenderer.invoke('wp-list-media', { url, key, secret, page: 1, 
perPage: 1, mediaType: 'image' })       : Promise.resolve({ success: false }),
      url ? ipcRenderer.invoke('wp-list-media', { url, key, secret, page: 1, 
perPage: 1, mediaType: 'video' })       : Promise.resolve({ success: false }),
      url ? ipcRenderer.invoke('wp-list-media', { url, key, secret, page: 1, 
perPage: 1, mediaType: 'application' }) : Promise.resolve({ success: false })
    ]);

    // GCS
    if (gcsRes.status === 'fulfilled' && gcsRes.value && gcsRes.value.success) 
{
      const items = gcsRes.value.data.items || [];
      const more  = gcsRes.value.data.nextPageToken ? '+' : '';
      setStat('ms-stat-gcs-total', items.length + more);
      setStat('ms-stat-gcs-img',   items.filter(i => 
/^image\//i.test(i.contentType)).length + more, 'var(--teal)');
      setStat('ms-stat-gcs-vid',   items.filter(i => 
/^video\//i.test(i.contentType)).length, 'var(--blue)');
      setStat('ms-stat-gcs-doc',   items.filter(i =>
        /^application\//i.test(i.contentType) ||
        /\.(pdf|doc|docx|xls|xlsx|ppt|pptx|csv)$/i.test(i.name)
      ).length, 'var(--gold)');
      gcsBuildFolderTree(items);  // auto-populate folder tree
    } else {
      ['ms-stat-gcs-total','ms-stat-gcs-img','ms-stat-gcs-vid','ms-stat-gcs-doc
'].forEach(id => setStat(id, 'err', 'var(--red)'));
    }

    // WP
    const wi = (wpImgRes.status === 'fulfilled' && wpImgRes.value && 
wpImgRes.value.success) ? wpImgRes.value.data.total : '\u2014';
    const wv = (wpVidRes.status === 'fulfilled' && wpVidRes.value && 
wpVidRes.value.success) ? wpVidRes.value.data.total : '\u2014';
    const wd = (wpDocRes.status === 'fulfilled' && wpDocRes.value && 
wpDocRes.value.success) ? wpDocRes.value.data.total : '\u2014';
    setStat('ms-stat-wp-img', wi, typeof wi === 'number' ? 'var(--blue)' : 
null);
    setStat('ms-stat-wp-vid', wv, typeof wv === 'number' ? 'var(--blue)' : 
null);
    setStat('ms-stat-wp-doc', wd, typeof wd === 'number' ? 'var(--gold)' : 
null);

    if (statsTag) statsTag.textContent = 'Updated ' + new 
Date().toLocaleTimeString();
    if (statsStatus) statsStatus.textContent = '';
    statsRefreshBtn.disabled = false;
    statsRefreshBtn.textContent = '\u21bb Refresh Counts';
  }

  if (statsRefreshBtn) statsRefreshBtn.addEventListener('click', statsLoad);

  // -------------------------------------------------------------
  //  GCS FOLDER TREE
  // -------------------------------------------------------------
  const gcsFolderToggleBtn = 
document.getElementById('ms-gcs-folder-toggle-btn');
  const gcsFolderPanel     = document.getElementById('ms-gcs-folder-panel');
  const gcsFolderList      = document.getElementById('ms-gcs-folder-list');
  const gcsFolderHint      = document.getElementById('ms-gcs-folder-hint');

  if (gcsFolderToggleBtn && gcsFolderPanel) {
    gcsFolderToggleBtn.addEventListener('click', () => {
      const visible = gcsFolderPanel.style.display !== 'none';
      gcsFolderPanel.style.display = visible ? 'none' : '';
      gcsFolderToggleBtn.textContent = visible ? '\uD83D\uDCC2 Folders' : 
'\uD83D\uDCC2 Hide Folders';
      // Auto-populate on first open
      if (!visible && gcsFolderList && !gcsFolderList.children.length) {
        if (gcsItems.length) gcsBuildFolderTree(gcsItems);
        else statsLoad();  // will also build tree from fetched items
      }
    });
  }

  function gcsBuildFolderTree(items) {
    if (!gcsFolderList) return;
    // Collect all 1-level and 2-level virtual folder paths from object names
    const folderSet = new Set(['']);  // '' = All Files (root)
    for (const item of items) {
      const parts = item.name.split('/').filter(Boolean);
      if (parts.length >= 2) {
        folderSet.add(parts[0] + '/');                        // e.g. "media/"
        if (parts.length >= 3) {
          folderSet.add(parts[0] + '/' + parts[1] + '/');    // e.g. 
"media/images/"
        }
      }
    }
    const sorted = [...folderSet].sort((a, b) => {
      if (a === '') return -1; if (b === '') return 1;
      return a.localeCompare(b);
    });

    gcsFolderList.innerHTML = '';
    if (gcsFolderHint) gcsFolderHint.style.display = sorted.length > 1 ? 
'none' : '';

    const prefixInput = document.getElementById('ms-gcs-prefix');
    for (const folder of sorted) {
      const depth = (folder.match(/\//g) || []).length;  // '' depth=0, 
'media/' depth=1, 'media/images/' depth=2
      const isL2  = depth === 2;
      const icon  = folder === '' ? '\uD83D\uDDC2' : isL2 ? '\uD83D\uDCC4' : 
'\uD83D\uDCC1';
      const label = folder === '' ? 'All Files' : folder;

      const chip = document.createElement('button');
      chip.className = 'btn btn-default btn-sm';
      chip.style.cssText = 'font-size:11px;padding:3px 
12px;border-radius:20px;' + (
        folder === '' ? 'color:var(--teal);border-color:rgba(90,200,250,0.4);' 
:
        isL2          ? 
'color:var(--gold);border-color:rgba(255,200,50,0.4);margin-left:14px;' :
                        'color:var(--fg-2);'
      );
      chip.textContent = icon + ' ' + label;
      chip.title = folder === '' ? 'Show all files' : 'Browse folder: ' + 
folder;
      chip.addEventListener('click', () => {
        if (prefixInput) prefixInput.value = folder;
        if (gcsLoadBtn) gcsLoadBtn.click();
      });
      gcsFolderList.appendChild(chip);
    }
  }

  // -------------------------------------------------------------
  //  GCS CREATE FOLDER
  // -------------------------------------------------------------
  const gcsFolderPreset  = document.getElementById('ms-gcs-folder-preset');
  const gcsFolderCustom  = document.getElementById('ms-gcs-folder-custom');
  const gcsMkFolderBtn   = document.getElementById('ms-gcs-mkfolder-btn');
  const gcsMkFolderRes   = document.getElementById('ms-gcs-mkfolder-result');

  if (gcsFolderPreset) {
    gcsFolderPreset.addEventListener('change', () => {
      if (gcsFolderPreset.value === 'custom') {
        gcsFolderCustom.style.display = '';
        gcsFolderCustom.focus();
      } else {
        gcsFolderCustom.style.display = 'none';
      }
    });
  }

  if (gcsMkFolderBtn) {
    gcsMkFolderBtn.addEventListener('click', async () => {
      const path = gcsFolderPreset && gcsFolderPreset.value === 'custom'
        ? (gcsFolderCustom ? gcsFolderCustom.value.trim() : '')
        : (gcsFolderPreset ? gcsFolderPreset.value : '');
      if (!path || path === '') {
        gcsMkFolderRes.style.color = 'var(--red)';
        gcsMkFolderRes.textContent = 'Select or type a folder path first';
        return;
      }
      gcsMkFolderBtn.disabled = true;
      gcsMkFolderBtn.textContent = 'Creating\u2026';
      gcsMkFolderRes.textContent = '';
      const r = await ipcRenderer.invoke('gcs-create-folder', { gcsConfig: 
gcsConf(), folderPath: path });
      if (r.success) {
        gcsMkFolderRes.style.color = 'var(--green)';
        gcsMkFolderRes.textContent = '\u2713 Created: ' + r.data.folder;
        // add chip to folder tree immediately
        if (gcsFolderList) {
          const prefixInput = document.getElementById('ms-gcs-prefix');
          const chip = document.createElement('button');
          chip.className = 'btn btn-default btn-sm';
          chip.style.cssText = 'font-size:11px;padding:3px 
12px;border-radius:20px;color:var(--teal);border-color:rgba(90,200,250,0.4);';
          chip.textContent = '\uD83D\uDCC1 ' + r.data.folder;
          chip.title = 'Browse folder: ' + r.data.folder;
          chip.addEventListener('click', () => {
            if (prefixInput) prefixInput.value = r.data.folder;
            if (gcsLoadBtn) gcsLoadBtn.click();
          });
          gcsFolderList.appendChild(chip);
          if (gcsFolderHint) gcsFolderHint.style.display = 'none';
        }
        toast('\u2713 GCS folder created: ' + r.data.folder, 'ok');
        // reset preset
        if (gcsFolderPreset) gcsFolderPreset.value = '';
        if (gcsFolderCustom) { gcsFolderCustom.value = ''; 
gcsFolderCustom.style.display = 'none'; }
      } else {
        gcsMkFolderRes.style.color = 'var(--red)';
        gcsMkFolderRes.textContent = '\u2717 ' + (r.error || 'Failed');
        toast('Folder create failed: ' + r.error, 'err');
      }
      gcsMkFolderBtn.disabled = false;
      gcsMkFolderBtn.textContent = '\uD83D\uDCC2 Create Folder';
    });
  }

})();

logRenderer('Calling populateSettings...');

    logRenderer('populateSettings failed: ' + e.message);

cfg0.azEndpoint ? 'INFO' : 'DEBUG');
sysLog(cfg0.gcsBucket ? `GCS bucket: ${cfg0.gcsBucket}` : 'GCS: not 
configured', cfg0.gcsBucket ? 'INFO' : 'DEBUG');
sysLog('Ready — select a module to begin', 'INFO');
logRenderer('Startup complete.');
